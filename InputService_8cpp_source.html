97 // Joystick axis event
98 void InputService::onJoystickEvent(const SDL_JoyAxisEvent event) {
99  // Translated axis
100  const int trans_axis = (JOY_MAX_AXES + event.axis) * 2;
101 
102  // Check if moved enough
103  if (event.value > JOYSTICK_DEADZONE) {
104  joystick_state.stick[trans_axis] = true;
105  } else if (event.value < -JOYSTICK_DEADZONE) {
106  joystick_state.stick[trans_axis + 1] = true;
107  } else {
108  joystick_state.stick[trans_axis] = false;
109  joystick_state.stick[trans_axis + 1] = false;
110  }
111 }
