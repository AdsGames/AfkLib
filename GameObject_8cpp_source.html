50 void GameObject::updateInternal() {
51  // Update delta position
52  last_x = x;
53  last_y = y;
54 
55  // Parent functions
56  if (parent_id != 0) {
57  // Autoremove if parent is dead
58  try {
59  auto& parent = scene.get<GameObject>(parent_id);
60 
61  // Set hooked state
62  hooked = parent.getHooked();
63 
64  // Set position
65  x += parent.x - parent.last_x;
66  y += parent.y - parent.last_y;
67 
68  } catch (const KeyLookupException&) {
69  scene.remove(id);
70  }
71  }
72 
73  // Colliders
74  std::remove_if(colliders.begin(), colliders.end(),
75  [&](const auto& obj_id) { return !scene.has(obj_id); });
76  for (auto& obj_id : colliders) {
77  collide(scene.get(obj_id));
78  }
79 }
